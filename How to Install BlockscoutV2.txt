# 1. Download all required .deb packages

wget https://download.docker.com/linux/ubuntu/dists/jammy/pool/stable/amd64/docker-ce_20.10.24~3-0~ubuntu-jammy_amd64.deb

# 2. Install in the correct dependency order
sudo dpkg -i containerd.io_1.6.28-2_amd64.deb
sudo dpkg -i docker-ce-cli_20.10.24~3-0~ubuntu-jammy_amd64.deb
sudo dpkg -i docker-ce_20.10.24~3-0~ubuntu-jammy_amd64.deb
sudo dpkg -i docker-compose-plugin_2.20.2-1~ubuntu.22.04~jammy_amd64.deb



"PLEASE DONT RENAME ANYTHING IN THE PROXY DIRECTORY, JUST MAKE CHANGES IN THE .tempaltes AND IT WILL WORK"



# Check Docker version
docker --version

# Check Docker Compose plugin version
docker compose version

# Create the 'docker' group
sudo groupadd docker

# Start the Docker service again
sudo systemctl daemon-reexec
sudo systemctl restart docker

# Verify Docker is running
sudo systemctl status docker

# Run test container (optional)
sudo docker run hello-world
